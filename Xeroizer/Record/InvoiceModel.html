def pdf(id, filename = nil)
  pdf_data = @application.http_get(@application.client, "#{url}/#{CGI.escape(id)}", :response => :pdf)
  if filename
    File.open(filename, "w") { | fp | fp.write pdf_data }
    nil
  else
    pdf_data
  end
end
