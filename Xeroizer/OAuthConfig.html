def load yaml_text
  require "yaml"
  yaml = YAML.load yaml_text
  consumer_credential = yaml["consumer"]

  OAuthCredentials.new(
    consumer_credential["key"],
    consumer_credential["secret"],
    consumer_credential["key_file"]
  )
end
